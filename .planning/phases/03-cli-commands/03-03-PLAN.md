   public interface IPortAllocator

       Task ReleasePortAsync(int port); // Placeholder for future port pooling
   }

2. **Implement PortAllocator:**
   - **Port range:** 4000-4999 (1000 ports available)
   - **Detection strategy:** TCP listener binding (reliable, prevents conflicts)
   - **AssignFreePortAsync implementation:**
     ```csharp
     public async Task<int> AssignFreePortAsync()
     {
         var random = new Random();
         const int maxAttempts = 50; // Prevent infinite loop
         int attempt = 0;

         while (attempt < maxAttempts)
         {
             // Generate random port in range
             var port = random.Next(4000, 5000);

             if (await IsPortFreeAsync(port))
             {
                 return port;
             }

             attempt++;
         }

         throw new InvalidOperationException("Failed to allocate port after 50 attempts. Port range 4000-4999 may be exhausted.");
     }
     ```

   - **IsPortFreeAsync implementation:**
     ```csharp
     public async Task<bool> IsPortFreeAsync(int port)
     {
         try
         {
             using var listener = new TcpListener(IPAddress.Loopback, port);
             listener.Start();
             listener.Stop();
             return true; // Port is free if we can bind to it
         }
         catch (SocketException) when (port >= 4000 && port <= 4999)
         {
             return false; // Port is in use
         }
     }
     ```

   - **ReleasePortAsync implementation:**
     ```csharp
     public Task ReleasePortAsync(int port)
     {
         // Placeholder for future port pooling
         // Currently no-op since ports are not tracked after allocation
         return Task.CompletedTask;
     }
     ```

DO NOT:
- Use sequential port scanning (too slow, 4000, 4001, 4002...)
- Use OS port randomization (can cause conflicts)
- Return ports outside 4000-4999 range
- Implement port pooling in this task (future enhancement)

2. **Create RunSettings.cs:**
   ```csharp
   using Spectre.Console.Cli;

   public class RunSettings : CommandSettings
   {
       [CommandArgument(0, "[NAME]")]
       [Description("Name for the route (e.g., 'api' creates api.localhost)")]
       public string Name { get; set; } = string.Empty;

       [CommandArgument(1, "[COMMAND...]")]
       [Description("Command to run with PORT variable injected")]
       public string[] Command { get; set; } = Array.Empty<string>();

       public override ValidationResult Validate()
       {
           // Validate name
           if (string.IsNullOrWhiteSpace(Name))
           {
               return ValidationResult.Error("Name is required");
           }

           if (Name.Contains(' ') || Name.Contains('/') || Name.Contains('\\'))
           {
               return ValidationResult.Error("Name must be a valid hostname (no spaces or slashes)");
           }

           // Validate command
           if (Command.Length == 0)
           {
               return ValidationResult.Error("Command is required");
           }

           // Check for duplicate routes
           var hostname = $"{Name}.localhost";
           var routes = _routeStore.LoadRoutesAsync().GetAwaiter().GetResult();
           if (routes.Any(r => r.Hostname == hostname))
           {
               return ValidationResult.Error($"Route '{Name}' already exists. Use 'portless list' to see active routes.");
           }

           return ValidationResult.Success();
       }
   }
   ```
