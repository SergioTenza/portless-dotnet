  <action>
    Add X-Forwarded headers configuration to YARP setup in Portless.Proxy/Program.cs (line 54-56):

    Current code:

    ```csharp
        .LoadFromMemory([],[]);
    ```

    Change to:
    ```csharp
    builder.Services.AddReverseProxy()
        .LoadFromMemory([],[])
        .AddTransforms<ForwardedTransformFactory>(); // Add X-Forwarded-* headers

    // Transform factory for X-Forwarded headers
    public class ForwardedTransformFactory : ITransformProvider
    {
        public void ValidateRoute(RouteConfig route, IList<string> errors)
        {
            // No validation needed - this transform applies to all routes
        }

        public void ValidateCluster(ClusterConfig cluster, IList<string> errors)
        {
            // No validation needed
        }

        public void Apply(TransformBuilderContext context)
        {
            // Add X-Forwarded-* headers (For, Proto, Host)
            context.AddRequestTransform(async transformContext =>
            {
                var originalHost = transformContext.HttpContext.Request.Host.Value;
                var originalProto = transformContext.HttpContext.Request.Scheme;

                transformContext.ProxyRequest.Headers.TryAdd("X-Forwarded-Host", originalHost);
                transformContext.ProxyRequest.Headers.TryAdd("X-Forwarded-Proto", originalProto);
                transformContext.ProxyRequest.Headers.TryAdd("X-Forwarded-For",
                    transformContext.HttpContext.Connection.RemoteIpAddress?.ToString() ?? "unknown");

                // Preserve original protocol information
                var protocol = transformContext.HttpContext.Request.Protocol;
                transformContext.ProxyRequest.Headers.TryAdd("X-Forwarded-Protocol", protocol);

                await Task.CompletedTask;
            });
        }
    }
    ```

    Place the ForwardedTransformFactory class at the end of Program.cs (after AddHostRequest record).

    This configuration:
    - Preserves original client information in forwarded headers
    - Adds X-Forwarded-Host, X-Forwarded-Proto, X-Forwarded-For
    - Adds X-Forwarded-Protocol to distinguish HTTP/1.1 vs HTTP/2
    - Maintains backward compatibility with PROTO-05 requirement
