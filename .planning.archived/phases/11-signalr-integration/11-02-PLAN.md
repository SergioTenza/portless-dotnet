```csharp
[Fact]
public async Task SignalR_Connection_Established_Through_Proxy()
{
    // Arrange
    var hubUrl = GetHubUrl(_factory, "/chathub");
    var connection = new HubConnectionBuilder()
        .WithUrl(hubUrl)
        .Build();

    // Act
    await connection.StartAsync();

    // Assert
    Assert.Equal(HubConnectionState.Connected, connection.State);

    // Cleanup
    await connection.StopAsync();
}

private string GetHubUrl(WebApplicationFactory<Program> factory, string path)
{
    var serverUrl = factory.Server.BaseAddress.ToString().TrimEnd('/');
    return $"{serverUrl}{path}";
}
```

```csharp
[Fact]
public async Task SignalR_Message_Sent_And_Received_Through_Proxy()
{
    // Arrange
    var hubUrl = GetHubUrl(_factory, "/chathub");
    var connection = new HubConnectionBuilder()
        .WithUrl(hubUrl)
        .Build();

    var tcs = new TaskCompletionSource<(string user, string message)>();
    connection.On<string, string>("ReceiveMessage", (user, message) =>
    {
        tcs.SetResult((user, message));
    });

    await connection.StartAsync();

    // Act
    await connection.InvokeAsync("SendMessage", "TestUser", "TestMessage");

    // Assert
    var received = await tcs.Task.WaitAsync(TimeSpan.FromSeconds(5));
    Assert.Equal("TestUser", received.user);
    Assert.Equal("TestMessage", received.message);

    // Cleanup
    await connection.StopAsync();
}
```

## Risk Analysis

**Risk 1: SignalR negotiation fails through proxy**
- **Impact:** Tests fail, can't verify SignalR works
- **Mitigation:** Ensure X-Forwarded headers are set (Phase 9), WebSocket support enabled (Phase 10)
- **Verification:** Run connection test first, verify negotiation succeeds

**Risk 2: Tests timeout waiting for messages**
- **Impact:** Tests fail or hang indefinitely
- **Mitigation:** Use TaskCompletionSource with timeout (5 seconds), add timeout to all async operations
- **Verification:** Run tests with intentional delays, ensure they fail gracefully

**Risk 3: Connection state is not stable**
- **Impact:** Flaky tests that sometimes pass, sometimes fail
- **Mitigation:** Always verify connection State before sending messages, use proper async/await
- **Verification:** Run tests multiple times to ensure stability

**Risk 4: Multiple tests interfere with each other**
- **Impact:** Tests pass individually but fail when run together
- **Mitigation:** Each test creates its own connection, proper cleanup with StopAsync()
- **Verification:** Run entire test suite, ensure no cross-test interference
